CREATE TABLE IF NOT EXISTS Genre (
id SERIAL PRIMARY KEY,
title VARCHAR(20) NOT NULL
);

CREATE TABLE IF NOT EXISTS Artist (
id SERIAL PRIMARY KEY,
nickname VARCHAR(30) NOT NULL
);

CREATE TABLE IF NOT EXISTS Mix (
genre_id INTEGER REFERENCES Genre(id),
artist_id INTEGER REFERENCES Artist(id),
CONSTRAINT pk PRIMARY KEY(genre_id, artist_id)
);

CREATE TABLE IF NOT EXISTS Album (
id SERIAL PRIMARY KEY,
title VARCHAR(25) NOT NULL,
year VARCHAR (4) NOT NULL
)

CREATE TABLE IF NOT EXISTS Mix_1 (
album_id INTEGER REFERENCES Album(id),
artist_id INTEGER REFERENCES Artist(id),
CONSTRAINT qk PRIMARY KEY(album_id, artist_id)
);

CREATE TABLE IF NOT EXISTS Song (
id INTEGER PRIMARY KEY REFERENCES Album(id),
name VARCHAR(25) NOT NULL,
duration VARCHAR(4) NOT NULL
);

CREATE TABLE IF NOT EXISTS Collection( 
id SERIAL PRIMARY KEY,
title VARCHAR(30) NOT NULL,
year VARCHAR(4) NOT NULL
);

CREATE TABLE IF NOT EXISTS Mix_2(
collection_id INTEGER REFERENCES Collection(id),
song_id INTEGER REFERENCES Song(id),
CONSTRAINT jk PRIMARY KEY(collection_id, song_id)
)
